

#include <iostream>
using namespace std;
class Area {
  public:
    double calculateArea(double length, double breadth) {
      return length * breadth;
    }

    double calculateArea(double side) {
      return side * side;
    }
};

int main(void) {
  Area area;
  cout << "Area of rectangle = " << area.calculateArea(3, 4) << endl;
  cout << "Area of square = " << area.calculateArea(6) << endl;
  return 0;
}
